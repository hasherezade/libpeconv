66{

72 return nullptr;

74 DWORD r_size = GetFileSize(file, 0);
75 if (read_size != 0 && read_size <= r_size) {
76 r_size = MASK_TO_DWORD(read_size);

78 peconv::UNALIGNED_BUF buffer = peconv::alloc_unaligned(r_size);

82#endif
83 return nullptr;
84 }
85 DWORD out_size = 0;

87#ifdef _DEBUG
88 std::cerr << "Reading failed!" << std::endl;
89#endif
90 peconv::free_file(buffer);
91 buffer = nullptr;
92 read_size = 0;
93 } else {
94 read_size = r_size;
95 }
