21 template <typename INT_TYPE>
22 INT_TYPE round_up_to_unit(const INT_TYPE size, const INT_TYPE unit)
23 {
24 if (unit == 0) return size;
25
26 INT_TYPE rem = size % unit;
27 if (rem == 0) return size;
28
29 return size + (unit - rem);
30 }
