34BYTE* peconv::load_pe_module(BYTE* dllRawData, size_t r_size, OUT size_t &v_size, bool executable, bool relocate)
35{
36 if (!peconv::get_nt_hdrs(dllRawData, r_size)) {
37 return NULL;
38 }
39 if (peconv::get_sections_count(dllRawData, r_size) == 0) {
40 return load_no_sec_pe(dllRawData, r_size, v_size, executable);
41 }
42 // by default, allow to load the PE at any base:
43 ULONGLONG desired_base = NULL;
44 // if relocating is required, but the PE has no relocation table...
45 if (relocate && !has_relocations(dllRawData)) {
46 // ...enforce loading the PE image at its default base (so that it will need no relocations)
47 desired_base = get_image_base(dllRawData);
48 }
49 // load a virtual image of the PE file at the desired_base address (random if desired_base is NULL):
50 BYTE *mappedDLL = pe_raw_to_virtual(dllRawData, r_size, v_size, executable, desired_base);
51 if (mappedDLL) {
52 //if the image was loaded at its default base, relocate_module will return always true (because relocating is already done)
53 if (relocate && !relocate_module(mappedDLL, v_size, (ULONGLONG)mappedDLL)) {
54 // relocating was required, but it failed - thus, the full PE image is useless
55 std::cerr << "[!] Could not relocate the module!\n";
56 free_pe_buffer(mappedDLL, v_size);
57 mappedDLL = NULL;
58 }
59 } else {
60 std::cerr << "[!] Could not allocate memory at the desired base!\n";
61 }
62 return mappedDLL;
63}
